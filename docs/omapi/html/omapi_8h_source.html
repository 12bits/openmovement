00933                 | (((OM_DATETIME)(day)          & 0x1f) << 17) \
00934                 | (((OM_DATETIME)(hours)        & 0x1f) << 12) \
00935                 | (((OM_DATETIME)(minutes)      & 0x3f) <<  6) \
00936                 | (((OM_DATETIME)(seconds)      & 0x3f)      ) \
00937                 )
