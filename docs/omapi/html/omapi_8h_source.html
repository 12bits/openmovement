00751                 ( (((OM_DATETIME)((year) % 100) & 0x3f) << 26) \
00752                 | (((OM_DATETIME)(month)        & 0x0f) << 22) \
00753                 | (((OM_DATETIME)(day)          & 0x1f) << 17) \
00754                 | (((OM_DATETIME)(hours)        & 0x1f) << 12) \
00755                 | (((OM_DATETIME)(minutes)      & 0x3f) <<  6) \
00756                 | (((OM_DATETIME)(seconds)      & 0x3f)      ) \
00757                 )
