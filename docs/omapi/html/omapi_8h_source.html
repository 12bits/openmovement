00973 #define OM_DATETIME_FROM_YMDHMS(year, month, day, hours, minutes, seconds) \
00974                 ( (((OM_DATETIME)((year) % 100) & 0x3f) << 26) \
00975                 | (((OM_DATETIME)(month)        & 0x0f) << 22) \
00976                 | (((OM_DATETIME)(day)          & 0x1f) << 17) \
00977                 | (((OM_DATETIME)(hours)        & 0x1f) << 12) \
00978                 | (((OM_DATETIME)(minutes)      & 0x3f) <<  6) \
00979                 | (((OM_DATETIME)(seconds)      & 0x3f)      ) \
00980                 )
00981 #define OM_DATETIME_YEAR(dateTime)    ((unsigned int)((unsigned char)(((dateTime) >> 26) & 0x3f)) + 2000) 
00982 #define OM_DATETIME_MONTH(dateTime)   ((unsigned char)(((dateTime) >> 22) & 0x0f))  
00983 #define OM_DATETIME_DAY(dateTime)     ((unsigned char)(((dateTime) >> 17) & 0x1f))  
00984 #define OM_DATETIME_HOURS(dateTime)   ((unsigned char)(((dateTime) >> 12) & 0x1f))  
00985 #define OM_DATETIME_MINUTES(dateTime) ((unsigned char)(((dateTime) >>  6) & 0x3f))  
00986 #define OM_DATETIME_SECONDS(dateTime) ((unsigned char)(((dateTime)      ) & 0x3f))  
00987 #define OM_DATETIME_MIN_VALID OM_DATETIME_FROM_YMDHMS(2000,  1,  1,  0, 0,   0)     
00988 #define OM_DATETIME_MAX_VALID OM_DATETIME_FROM_YMDHMS(2063, 12, 31, 23, 59, 59)
